rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Função helper para verificar se é admin
    function isAdmin() {
      return request.auth != null && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['Presidenta', 'Vice-Presidente'];
    }
    
    // Função helper para verificar se é ministro
    function isMinister() {
      return request.auth != null && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['Ministro', 'Ministra', 'Advogado-Geral'];
    }
    
    // Collection de usuários
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create, delete: if false; // Criado apenas no setup
      allow update: if isAdmin(); // Admin pode atualizar (ex: adicionar habboUsername)
    }
    
    // Collection de orçamentos - admins editam, todos leem
    match /budgets/{budgetId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }
    
    // Collection de propostas
    match /proposals/{proposalId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null; // Qualquer um pode criar proposta
      allow update: if isAdmin(); // Apenas Presidenta/Vice aprovam/rejeitam
      allow delete: if isAdmin(); // Apenas Presidenta/Vice podem deletar (e reverter orçamento)
    }
    
    // Collection de notificações
    match /notifications/{notificationId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if isAdmin();
    }
  }
}
